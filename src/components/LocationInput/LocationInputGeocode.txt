import React, { useState, useEffect } from 'react';
import AutocompleteInput from './AutoCompleteInput';

const LocationInput: React.FC = () => {
  const [inputValue, setInputValue] = useState<string>('');
  const [suggestions, setSuggestions] = useState<string[]>([]);

  useEffect(() => {
    // Fetch valid locations based on inputValue using the Google Geocoding API
    const fetchGoogleLocations = async () => {
      try {
        const apiKey = 'AIzaSyB03z0J0Z5A9Xxv98hgTN26-DKL9FLw4p0';
        const response = await fetch(
          `https://maps.googleapis.com/maps/api/geocode/json?address=${inputValue}&key=${apiKey}`
        );

        if (!response.ok) {
          throw new Error('Request to Google Geocoding API failed');
        }

        const data = await response.json();
        const validLocations = data.results.map((item: any) => item.formatted_address);
        setSuggestions(validLocations);
        console.log("Input data is ",inputValue);
        console.log(data);
      } catch (error) {
        console.error('Error fetching Google Geocoding data:', error);
      }
    };

    if (inputValue && inputValue.trim().length > 2) {
      fetchGoogleLocations();
    } else {
      setSuggestions([]);
    }
  }, [inputValue]);

  const handleInputChange = (value: string) => {
    setInputValue(value);
  };

  return (
    <AutocompleteInput suggestions={suggestions} onInputChange={handleInputChange} />
  );
};

export default LocationInput;
